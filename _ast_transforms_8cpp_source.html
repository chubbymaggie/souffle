25 void ResolveAliasesTransformer::resolveAliases(AstProgram& program) {
26  // get all clauses
27  std::vector<const AstClause*> clauses;
28  visitDepthFirst(program, [&](const AstRelation& rel) {
29  for (const auto& cur : rel.getClauses()) {
30  clauses.push_back(cur);
31  }
32  });
33 
34  // clean all clauses
35  for (const AstClause* cur : clauses) {
36  // -- Step 1 --
37  // get rid of aliases
38  std::unique_ptr<AstClause> noAlias = resolveAliases(*cur);
39 
40  // clean up equalities
41  std::unique_ptr<AstClause> cleaned = removeTrivialEquality(*noAlias);
42 
43  // -- Step 2 --
44  // restore simple terms in atoms
45  removeComplexTermsInAtoms(*cleaned);
46 
47  // exchange rule
48  program.removeClause(cur);
49  program.appendClause(std::move(cleaned));
50  }
51 }

55 /**
56  * A utility class for the unification process required to eliminate
57  * aliases. A substitution maps variables to terms and can be applied
58  * as a transformation to AstArguments.
59  */
60 class Substitution {
61  // the type of map for storing mappings internally
62  // - variables are identified by their name (!)
63  typedef std::map<std::string, std::unique_ptr<AstArgument>> map_t;
64 
65  /** The mapping of variables to terms (see type def above) */
66  map_t map;
67 
68 public:
69  // -- Ctors / Dtors --
70 
71  Substitution(){};
72 
73  Substitution(const std::string& var, const AstArgument* arg) {
74  map.insert(std::make_pair(var, std::unique_ptr<AstArgument>(arg->clone())));
75  }
76 
77  virtual ~Substitution() = default;
78 
79  /**
80  * Applies this substitution to the given argument and
81  * returns a pointer to the modified argument.
82  *
83  * @param node the node to be transformed
84  * @return a pointer to the modified or replaced node
85  */
86  virtual std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const {
87  // create a substitution mapper
88  struct M : public AstNodeMapper {
89  const map_t& map;
90 
91  M(const map_t& map) : map(map) {}
92 
93  using AstNodeMapper::operator();
94 
95  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const override {
96  // see whether it is a variable to be substituted
97  if (auto var = dynamic_cast<AstVariable*>(node.get())) {
98  auto pos = map.find(var->getName());
99  if (pos != map.end()) {
100  return std::unique_ptr<AstNode>(pos->second->clone());
101  }
102  }
103 
104  // otherwise - apply mapper recursively
105  node->apply(*this);
106  return node;
107  }
108  };
109 
110  // apply mapper
111  return M(map)(std::move(node));
112  }
113 
114  /**
115  * A generic, type consistent wrapper of the transformation
116  * operation above.
117  */
118  template <typename T>
119  std::unique_ptr<T> operator()(std::unique_ptr<T> node) const {
120  std::unique_ptr<AstNode> resPtr =
121  (*this)(std::unique_ptr<AstNode>(static_cast<AstNode*>(node.release())));
122  assert(dynamic_cast<T*>(resPtr.get()) && "Invalid node type mapping.");
123  return std::unique_ptr<T>(dynamic_cast<T*>(resPtr.release()));
124  }
125 
126  /**
127  * Appends the given substitution to this substitution such that
128  * this substitution has the same effect as applying this following
129  * the given substitution in sequence.
130  */
131  void append(const Substitution& s) {
132  // apply substitution on all current mappings
133  for (auto& cur : map) {
134  cur.second = s(std::move(cur.second));
135  }
136 
137  // append uncovered variables to the end
138  for (const auto& cur : s.map) {
139  auto pos = map.find(cur.first);
140  if (pos != map.end()) {
141  continue;
142  }
143  map.insert(std::make_pair(cur.first, std::unique_ptr<AstArgument>(cur.second->clone())));
144  }
145  }
146 
147  /** A print function (for debugging) */
148  void print(std::ostream& out) const {
149  out << "{" << join(map, ",",
150  [](std::ostream& out,
151  const std::pair<const std::string, std::unique_ptr<AstArgument>>& cur) {
152  out << cur.first << " -> " << *cur.second;
153  })
154  << "}";
155  }
156 
157  friend std::ostream& operator<<(std::ostream& out, const Substitution& s) __attribute__((unused)) {
158  s.print(out);
159  return out;
160  }

208 std::unique_ptr<AstClause> ResolveAliasesTransformer::resolveAliases(const AstClause& clause) {
209  /**
210  * This alias analysis utilizes unification over the equality
211  * constraints in clauses.
212  */
213 
214  // -- utilities --
215 
216  // tests whether something is a ungrounded variable
217  auto isVar = [&](const AstArgument& arg) { return dynamic_cast<const AstVariable*>(&arg); };
218 
219  // tests whether something is a record
220  auto isRec = [&](const AstArgument& arg) { return dynamic_cast<const AstRecordInit*>(&arg); };
221 
222  // tests whether a value a occurs in a term b
223  auto occurs = [](const AstArgument& a, const AstArgument& b) {
224  bool res = false;
225  visitDepthFirst(b, [&](const AstArgument& cur) { res = res || cur == a; });
226  return res;
227  };
228 
229  // I) extract equations
230  std::vector<Equation> equations;
231  visitDepthFirst(clause, [&](const AstConstraint& rel) {
232  if (rel.getOperator() == BinaryConstraintOp::EQ) {
233  equations.push_back(Equation(rel.getLHS(), rel.getRHS()));
234  }
235  });
236 
237  // II) compute unifying substitution
238  Substitution substitution;
239 
240  // a utility for processing newly identified mappings
241  auto newMapping = [&](const std::string& var, const AstArgument* term) {
242  // found a new substitution
243  Substitution newMapping(var, term);
244 
245  // apply substitution to all remaining equations
246  for (auto& cur : equations) {
247  cur.apply(newMapping);
248  }
249 
250  // add mapping v -> t to substitution
251  substitution.append(newMapping);
252  };
253 
254  while (!equations.empty()) {
255  // get next equation to compute
256  Equation cur = equations.back();
257  equations.pop_back();
258 
259  // shortcuts for left/right
260  const AstArgument& a = *cur.lhs;
261  const AstArgument& b = *cur.rhs;
262 
263  // #1: t = t => skip
264  if (a == b) {
265  continue;
266  }
267 
268  // #2: [..] = [..] => decompose
269  if (isRec(a) && isRec(b)) {
270  // get arguments
271  const auto& args_a = static_cast<const AstRecordInit&>(a).getArguments();
272  const auto& args_b = static_cast<const AstRecordInit&>(b).getArguments();
273 
274  // make sure sizes are identical
275  assert(args_a.size() == args_b.size());
276 
277  // create new equalities
278  for (size_t i = 0; i < args_a.size(); ++i) {
279  equations.push_back(Equation(args_a[i], args_b[i]));
280  }
281  continue;
282  }
283 
284  // neither is a variable
285  if (!isVar(a) && !isVar(b)) {
286  continue; // => nothing to do
287  }
288 
289  // both are variables
290  if (isVar(a) && isVar(b)) {
291  // a new mapping is found
292  auto& var = static_cast<const AstVariable&>(a);
293  newMapping(var.getName(), &b);
294  continue;
295  }
296 
297  // #3: t = v => swap
298  if (!isVar(a)) {
299  equations.push_back(Equation(b, a));
300  continue;
301  }
302 
303  // now we know a is a variable
304  assert(isVar(a));
305 
306  // we have v = t
307  const AstVariable& v = static_cast<const AstVariable&>(a);
308  const AstArgument& t = b;
309 
310  // #4: v occurs in t
311  if (occurs(v, t)) {
312  continue;
313  }
314 
315  assert(!occurs(v, t));
316 
317  // add new maplet
318  newMapping(v.getName(), &t);
319  }
320 
321  // III) compute resulting clause
322  return substitution(std::unique_ptr<AstClause>(clause.clone()));
323 }

325 std::unique_ptr<AstClause> ResolveAliasesTransformer::removeTrivialEquality(const AstClause& clause) {
326  // finally: remove t = t constraints
327  std::unique_ptr<AstClause> res(clause.cloneHead());
328  for (AstLiteral* cur : clause.getBodyLiterals()) {
329  // filter out t = t
330  if (AstConstraint* rel = dynamic_cast<AstConstraint*>(cur)) {
331  if (rel->getOperator() == BinaryConstraintOp::EQ) {
332  if (*rel->getLHS() == *rel->getRHS()) {
333  continue; // skip this one
334  }
335  }
336  }
337  res->addToBody(std::unique_ptr<AstLiteral>(cur->clone()));
338  }
339 
340  // done
341  return res;
342 }

344 void ResolveAliasesTransformer::removeComplexTermsInAtoms(AstClause& clause) {
345  // restore temporary variables for expressions in atoms
346 
347  // get list of atoms
348  std::vector<AstAtom*> atoms;
349  for (AstLiteral* cur : clause.getBodyLiterals()) {
350  if (AstAtom* arg = dynamic_cast<AstAtom*>(cur)) {
351  atoms.push_back(arg);
352  }
353  }
354 
355  // find all binary operations in atoms
356  std::vector<const AstArgument*> terms;
357  for (const AstAtom* cur : atoms) {
358  for (const AstArgument* arg : cur->getArguments()) {
359  // only interested in functions
360  if (!(dynamic_cast<const AstFunctor*>(arg))) {
361  continue;
362  }
363  // add this one if not yet registered
364  if (!any_of(terms, [&](const AstArgument* cur) { return *cur == *arg; })) {
365  terms.push_back(arg);
366  }
367  }
368  }
369 
370  // substitute them with variables (a real map would compare pointers)
371  typedef std::vector<std::pair<std::unique_ptr<AstArgument>, std::unique_ptr<AstVariable>>>
372  substitution_map;
373  substitution_map map;
374 
375  int var_counter = 0;
376  for (const AstArgument* arg : terms) {
377  map.push_back(std::make_pair(std::unique_ptr<AstArgument>(arg->clone()),
378  std::make_unique<AstVariable>(" _tmp_" + toString(var_counter++))));
379  }
380 
381  // apply mapping to replace terms with variables
382  struct Update : public AstNodeMapper {
383  const substitution_map& map;
384  Update(const substitution_map& map) : map(map) {}
385  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const override {
386  // check whether node needs to be replaced
387  for (const auto& cur : map) {
388  if (*cur.first == *node) {
389  return std::unique_ptr<AstNode>(cur.second->clone());
390  }
391  }
392  // continue recursively
393  node->apply(*this);
394  return node;
395  }
396  };
397 
398  Update update(map);
399 
400  // update atoms
401  for (AstAtom* atom : atoms) {
402  atom->apply(update);
403  }
404 
405  // add variable constraints to clause
406  for (const auto& cur : map) {
407  clause.addToBody(std::unique_ptr<AstLiteral>(
408  new AstConstraint(BinaryConstraintOp::EQ, std::unique_ptr<AstArgument>(cur.second->clone()),
409  std::unique_ptr<AstArgument>(cur.first->clone()))));
410  }
411 }

413 bool RemoveRelationCopiesTransformer::removeRelationCopies(AstProgram& program) {
414  typedef std::map<AstRelationIdentifier, AstRelationIdentifier> alias_map;
415 
416  // tests whether something is a variable
417  auto isVar = [&](const AstArgument& arg) { return dynamic_cast<const AstVariable*>(&arg); };
418 
419  // tests whether something is a record
420  auto isRec = [&](const AstArgument& arg) { return dynamic_cast<const AstRecordInit*>(&arg); };
421 
422  // collect aliases
423  alias_map isDirectAliasOf;
424 
425  // search for relations only defined by a single rule ..
426  for (AstRelation* rel : program.getRelations()) {
427  if (!rel->isComputed() && rel->getClauses().size() == 1u) {
428  // .. of shape r(x,y,..) :- s(x,y,..)
429  AstClause* cl = rel->getClause(0);
430  if (!cl->isFact() && cl->getBodySize() == 1u && cl->getAtoms().size() == 1u) {
431  AstAtom* atom = cl->getAtoms()[0];
432  if (equal_targets(cl->getHead()->getArguments(), atom->getArguments())) {
433  // we have a match but have to check that all arguments are either
434  // variables or records containing variables
435  bool onlyVars = true;
436  auto args = cl->getHead()->getArguments();
437  while (!args.empty()) {
438  const auto& cur = args.back();
439  args.pop_back();
440  if (!isVar(*cur)) {
441  if (isRec(*cur)) {
442  // records are decomposed and their arguments are checked
443  const auto& rec_args = static_cast<const AstRecordInit&>(*cur).getArguments();
444  for (size_t i = 0; i < rec_args.size(); ++i) {
445  args.push_back(rec_args[i]);
446  }
447  } else {
448  onlyVars = false;
449  break;
450  }
451  }
452  }
453  if (onlyVars) {
454  // all arguments are either variables or records containing variables
455  isDirectAliasOf[cl->getHead()->getName()] = atom->getName();
456  }
457  }
458  }
459  }
460  }
461 
462  // map each relation to its ultimate alias (could be transitive)
463  alias_map isAliasOf;
464 
465  // track any copy cycles; cyclic rules are effectively empty
466  std::set<AstRelationIdentifier> cycle_reps;
467 
468  for (std::pair<AstRelationIdentifier, AstRelationIdentifier> cur : isDirectAliasOf) {
469  // compute replacement
470 
471  std::set<AstRelationIdentifier> visited;
472  visited.insert(cur.first);
473  visited.insert(cur.second);
474 
475  auto pos = isDirectAliasOf.find(cur.second);
476  while (pos != isDirectAliasOf.end()) {
477  if (visited.count(pos->second)) {
478  cycle_reps.insert(cur.second);
479  break;
480  }
481  cur.second = pos->second;
482  pos = isDirectAliasOf.find(cur.second);
483  }
484  isAliasOf[cur.first] = cur.second;
485  }
486 
487  if (isAliasOf.empty()) {
488  return false;
489  }
490 
491  // replace usage of relations according to alias map
492  visitDepthFirst(program, [&](const AstAtom& atom) {
493  auto pos = isAliasOf.find(atom.getName());
494  if (pos != isAliasOf.end()) {
495  const_cast<AstAtom&>(atom).setName(pos->second);
496  }
497  });
498 
499  // break remaining cycles
500  for (const auto& rep : cycle_reps) {
501  auto rel = program.getRelation(rep);
502  rel->removeClause(rel->getClause(0));
503  }
504 
505  // remove unused relations
506  for (const auto& cur : isAliasOf) {
507  if (!cycle_reps.count(cur.first)) {
508  program.removeRelation(program.getRelation(cur.first)->getName());
509  }
510  }
511 
512  return true;
513 }

548 bool MaterializeAggregationQueriesTransformer::materializeAggregationQueries(
549  AstTranslationUnit& translationUnit) {
550  bool changed = false;
551 
552  AstProgram& program = *translationUnit.getProgram();
553  const TypeEnvironment& env = translationUnit.getAnalysis<TypeEnvironmentAnalysis>()->getTypeEnvironment();
554 
555  // if an aggregator has a body consisting of more than an atom => create new relation
556  int counter = 0;
557  visitDepthFirst(program, [&](const AstClause& clause) {
558  visitDepthFirstPostOrder(clause, [&](const AstAggregator& agg) {
559  // check whether a materialization is required
560  if (!needsMaterializedRelation(agg)) {
561  return;
562  }
563 
564  changed = true;
565 
566  // for more body literals: create a new relation
567  std::set<std::string> vars;
568  visitDepthFirst(agg, [&](const AstVariable& var) { vars.insert(var.getName()); });
569 
570  // -- create a new clause --
571 
572  auto relName = "__agg_rel_" + toString(counter++);
573  while (program.getRelation(relName) != nullptr) {
574  relName = "__agg_rel_" + toString(counter++);
575  }
576 
577  AstAtom* head = new AstAtom();
578  head->setName(relName);
579  std::vector<bool> symbolArguments;
580  for (const auto& cur : vars) {
581  head->addArgument(std::make_unique<AstVariable>(cur));
582  }
583 
584  AstClause* aggClause = new AstClause();
585  aggClause->setHead(std::unique_ptr<AstAtom>(head));
586  for (const auto& cur : agg.getBodyLiterals()) {
587  aggClause->addToBody(std::unique_ptr<AstLiteral>(cur->clone()));
588  }
589 
590  // instantiate unnamed variables in count operations
591  if (agg.getOperator() == AstAggregator::count) {
592  int count = 0;
593  for (const auto& cur : aggClause->getBodyLiterals()) {
594  cur->apply(
595  makeLambdaMapper([&](std::unique_ptr<AstNode> node) -> std::unique_ptr<AstNode> {
596  // check whether it is a unnamed variable
597  AstUnnamedVariable* var = dynamic_cast<AstUnnamedVariable*>(node.get());
598  if (!var) {
599  return node;
600  }
601 
602  // replace by variable
603  auto name = " _" + toString(count++);
604  auto res = new AstVariable(name);
605 
606  // extend head
607  head->addArgument(std::unique_ptr<AstArgument>(res->clone()));
608 
609  // return replacement
610  return std::unique_ptr<AstNode>(res);
611  }));
612  }
613  }
614 
615  // -- build relation --
616 
617  AstRelation* rel = new AstRelation();
618  rel->setName(relName);
619  // add attributes
620  std::map<const AstArgument*, TypeSet> argTypes =
621  TypeAnalysis::analyseTypes(env, *aggClause, &program);
622  for (const auto& cur : head->getArguments()) {
623  rel->addAttribute(std::unique_ptr<AstAttribute>(new AstAttribute(
624  toString(*cur), (isNumberType(argTypes[cur])) ? "number" : "symbol")));
625  }
626 
627  rel->addClause(std::unique_ptr<AstClause>(aggClause));
628  program.appendRelation(std::unique_ptr<AstRelation>(rel));
629 
630  // -- update aggregate --
631  AstAtom* aggAtom = head->clone();
632 
633  // count the usage of variables in the clause
634  // outside of aggregates. Note that the visitor
635  // is exhaustive hence double counting occurs
636  // which needs to be deducted for variables inside
637  // the aggregators and variables in the expression
638  // of aggregate need to be added. Counter is zero
639  // if the variable is local to the aggregate
640  std::map<std::string, int> varCtr;
641  visitDepthFirst(clause, [&](const AstArgument& arg) {
642  if (const AstAggregator* a = dynamic_cast<const AstAggregator*>(&arg)) {
643  visitDepthFirst(arg, [&](const AstVariable& var) { varCtr[var.getName()]--; });
644  if (a->getTargetExpression() != nullptr) {
645  visitDepthFirst(*a->getTargetExpression(),
646  [&](const AstVariable& var) { varCtr[var.getName()]++; });
647  }
648  } else {
649  visitDepthFirst(arg, [&](const AstVariable& var) { varCtr[var.getName()]++; });
650  }
651  });
652  for (size_t i = 0; i < aggAtom->getArity(); i++) {
653  if (AstVariable* var = dynamic_cast<AstVariable*>(aggAtom->getArgument(i))) {
654  // replace local variable by underscore if local
655  if (varCtr[var->getName()] == 0) {
656  aggAtom->setArgument(i, std::make_unique<AstUnnamedVariable>());
657  }
658  }
659  }
660  const_cast<AstAggregator&>(agg).clearBodyLiterals();
661  const_cast<AstAggregator&>(agg).addBodyLiteral(std::unique_ptr<AstLiteral>(aggAtom));
662  });
663  });
664 
665  return changed;
666 }

668 bool MaterializeAggregationQueriesTransformer::needsMaterializedRelation(const AstAggregator& agg) {
669  // everything with more than 1 body literal => materialize
670  if (agg.getBodyLiterals().size() > 1) {
671  return true;
672  }
673 
674  // inspect remaining atom more closely
675  const AstAtom* atom = dynamic_cast<const AstAtom*>(agg.getBodyLiterals()[0]);
676  if (!atom) {
677  // no atoms, so materialize
678  return true;
679  }
680 
681  // if the same variable occurs several times => materialize
682  bool duplicates = false;
683  std::set<std::string> vars;
684  visitDepthFirst(*atom,
685  [&](const AstVariable& var) { duplicates = duplicates | !vars.insert(var.getName()).second; });
686 
687  // if there are duplicates a materialization is required
688  if (duplicates) {
689  return true;
690  }
691 
692  // for all others the materialization can be skipped
693  return false;

723 void RemoveEmptyRelationsTransformer::removeEmptyRelationUses(
724  AstTranslationUnit& translationUnit, AstRelation* emptyRelation) {
725  AstProgram& program = *translationUnit.getProgram();
726 
727  //
728  // (1) drop rules from the program that have empty relations in their bodies.
729  // (2) drop negations of empty relations
730  //
731  // get all clauses
732  std::vector<const AstClause*> clauses;
733  visitDepthFirst(program, [&](const AstClause& cur) { clauses.push_back(&cur); });
734 
735  // clean all clauses
736  for (const AstClause* cl : clauses) {
737  // check for an atom whose relation is the empty relation
738 
739  bool removed = false;
740  ;
741  for (AstLiteral* lit : cl->getBodyLiterals()) {
742  if (AstAtom* arg = dynamic_cast<AstAtom*>(lit)) {
743  if (getAtomRelation(arg, &program) == emptyRelation) {
744  program.removeClause(cl);
745  removed = true;
746  break;
747  }
748  }
749  }
750 
751  if (!removed) {
752  // check whether a negation with empty relations exists
753 
754  bool rewrite = false;
755  for (AstLiteral* lit : cl->getBodyLiterals()) {
756  if (AstNegation* neg = dynamic_cast<AstNegation*>(lit)) {
757  if (getAtomRelation(neg->getAtom(), &program) == emptyRelation) {
758  rewrite = true;
759  break;
760  }
761  }
762  }
763 
764  if (rewrite) {
765  // clone clause without negation for empty relations
766 
767  auto res = std::unique_ptr<AstClause>(cl->cloneHead());
768 
769  for (AstLiteral* lit : cl->getBodyLiterals()) {
770  if (AstNegation* neg = dynamic_cast<AstNegation*>(lit)) {
771  if (getAtomRelation(neg->getAtom(), &program) != emptyRelation) {
772  res->addToBody(std::unique_ptr<AstLiteral>(lit->clone()));
773  }
774  } else {
775  res->addToBody(std::unique_ptr<AstLiteral>(lit->clone()));
776  }
777  }
778 
779  program.removeClause(cl);
780  program.appendClause(std::move(res));
781  }
782  }
783  }

800 bool ExtractDisconnectedLiteralsTransformer::transform(AstTranslationUnit& translationUnit) {
801  bool changed = false;
802  AstProgram& program = *translationUnit.getProgram();
803 
804  /* Process:
805  * Go through each clause and construct a variable dependency graph G.
806  * The nodes of G are the variables. A path between a and b exists iff
807  * a and b appear in a common body literal.
808  *
809  * Based on the graph, we can extract the body literals that are not associated
810  * with the arguments in the head atom into a new relation.
811  *
812  * E.g. a(x) :- b(x), c(y), d(y). will be transformed into:
813  * - a(x) :- b(x), newrel().
814  * - newrel() :- c(y), d(y).
815  *
816  * Note that only one pass through the clauses is needed:
817  * - All arguments in the body literals of the transformed clause cannot be
818  * independent of the head arguments (by construction).
819  * - The new relations holding the disconnected body literals have no
820  * head arguments by definition, and so the transformation does not apply.
821  */
822 
823  std::vector<AstClause*> clausesToAdd;
824  std::vector<const AstClause*> clausesToRemove;
825 
826  visitDepthFirst(program, [&](const AstClause& clause) {
827  // get head variables
828  std::set<std::string> headVars;
829  visitDepthFirst(*clause.getHead(), [&](const AstVariable& var) { headVars.insert(var.getName()); });
830 
831  // nothing to do if no arguments in the head
832  if (headVars.empty()) {
833  return;
834  }
835 
836  // construct the graph
837  Graph<std::string> variableGraph = Graph<std::string>();
838 
839  // add in its nodes
840  visitDepthFirst(clause, [&](const AstVariable& var) { variableGraph.insert(var.getName()); });
841 
842  // add in the edges
843  // since we are only looking at reachability, we can just add in an
844  // undirected edge from the first argument in the literal to each of the others
845 
846  // edges from the head
847  std::string firstVariable = *headVars.begin();
848  headVars.erase(headVars.begin());
849  for (const std::string& var : headVars) {
850  variableGraph.insert(firstVariable, var);
851  variableGraph.insert(var, firstVariable);
852  }
853 
854  // edges from literals
855  for (AstLiteral* bodyLiteral : clause.getBodyLiterals()) {
856  std::set<std::string> litVars;
857 
858  visitDepthFirst(*bodyLiteral, [&](const AstVariable& var) { litVars.insert(var.getName()); });
859 
860  // no new edges if only one variable is present
861  if (litVars.size() > 1) {
862  std::string firstVariable = *litVars.begin();
863  litVars.erase(litVars.begin());
864 
865  // create the undirected edge
866  for (const std::string& var : litVars) {
867  variableGraph.insert(firstVariable, var);
868  variableGraph.insert(var, firstVariable);
869  }
870  }
871  }
872 
873  // run a DFS from the first head variable
874  std::set<std::string> importantVariables;
875  variableGraph.visitDepthFirst(
876  firstVariable, [&](const std::string& var) { importantVariables.insert(var); });
877 
878  // partition the literals into connected and disconnected based on their variables
879  std::vector<AstLiteral*> connectedLiterals;
880  std::vector<AstLiteral*> disconnectedLiterals;
881 
882  for (AstLiteral* bodyLiteral : clause.getBodyLiterals()) {
883  bool connected = false;
884  bool hasArgs = false; // ignore literals with no arguments
885 
886  visitDepthFirst(*bodyLiteral, [&](const AstArgument& arg) {
887  hasArgs = true;
888 
889  if (auto var = dynamic_cast<const AstVariable*>(&arg)) {
890  if (importantVariables.find(var->getName()) != importantVariables.end()) {
891  connected = true;
892  }
893  }
894  });
895 
896  if (connected || !hasArgs) {
897  connectedLiterals.push_back(bodyLiteral);
898  } else {
899  disconnectedLiterals.push_back(bodyLiteral);
900  }
901  }
902 
903  if (!disconnectedLiterals.empty()) {
904  // need to extract some disconnected lits!
905  changed = true;
906 
907  static int disconnectedCount = 0;
908  std::stringstream nextName;
909  nextName << "+disconnected" << disconnectedCount;
910  AstRelationIdentifier newRelationName = nextName.str();
911  disconnectedCount++;
912 
913  // create the extracted relation and clause
914  // newrel() :- disconnectedLiterals(x).
915  auto newRelation = std::make_unique<AstRelation>();
916  newRelation->setName(newRelationName);
917  program.appendRelation(std::move(newRelation));
918 
919  AstClause* disconnectedClause = new AstClause();
920  disconnectedClause->setSrcLoc(clause.getSrcLoc());
921  disconnectedClause->setHead(std::make_unique<AstAtom>(newRelationName));
922 
923  for (AstLiteral* disconnectedLit : disconnectedLiterals) {
924  disconnectedClause->addToBody(std::unique_ptr<AstLiteral>(disconnectedLit->clone()));
925  }
926 
927  // create the replacement clause
928  // a(x) :- b(x), c(y). --> a(x) :- b(x), newrel().
929  AstClause* newClause = new AstClause();
930  newClause->setSrcLoc(clause.getSrcLoc());
931  newClause->setHead(std::unique_ptr<AstAtom>(clause.getHead()->clone()));
932 
933  for (AstLiteral* connectedLit : connectedLiterals) {
934  newClause->addToBody(std::unique_ptr<AstLiteral>(connectedLit->clone()));
935  }
936 
937  // add the disconnected clause to the body
938  newClause->addToBody(std::make_unique<AstAtom>(newRelationName));
939 
940  // replace the original clause with the new clauses
941  clausesToAdd.push_back(newClause);
942  clausesToAdd.push_back(disconnectedClause);
943  clausesToRemove.push_back(&clause);
944  }
945  });
946 
947  for (AstClause* newClause : clausesToAdd) {
948  program.appendClause(std::unique_ptr<AstClause>(newClause));
949  }
950 
951  for (const AstClause* oldClause : clausesToRemove) {
952  program.removeClause(oldClause);
953  }
954 
955  return changed;
956 }

958 bool ReduceExistentialsTransformer::transform(AstTranslationUnit& translationUnit) {
959  AstProgram& program = *translationUnit.getProgram();
960 
961  // Checks whether a given clause is recursive
962  auto isRecursiveClause = [&](const AstClause& clause) {
963  AstRelationIdentifier relationName = clause.getHead()->getName();
964  bool recursive = false;
965  visitDepthFirst(clause.getBodyLiterals(), [&](const AstAtom& atom) {
966  if (atom.getName() == relationName) {
967  recursive = true;
968  }
969  });
970  return recursive;
971  };
972 
973  // Checks whether an atom is of the form a(_,_,...,_)
974  auto isExistentialAtom = [&](const AstAtom& atom) {
975  for (AstArgument* arg : atom.getArguments()) {
976  if (!dynamic_cast<AstUnnamedVariable*>(arg)) {
977  return false;
978  }
979  }
980  return true;
981  };
982 
983  // Construct a dependency graph G where:
984  // - Each relation is a node
985  // - An edge (a,b) exists iff a uses b "non-existentially" in one of its *recursive* clauses
986  // This way, a relation can be transformed into an existential form
987  // if and only if all its predecessors can also be transformed.
988  Graph<AstRelationIdentifier> relationGraph = Graph<AstRelationIdentifier>();
989 
990  // Add in the nodes
991  for (AstRelation* relation : program.getRelations()) {
992  relationGraph.insert(relation->getName());
993  }
994 
995  // Keep track of all relations that cannot be transformed
996  std::set<AstRelationIdentifier> minimalIrreducibleRelations;
997 
998  for (AstRelation* relation : program.getRelations()) {
999  if (relation->isComputed() || relation->isInput()) {
1000  // No I/O relations can be transformed
1001  minimalIrreducibleRelations.insert(relation->getName());
1002  }
1003 
1004  for (AstClause* clause : relation->getClauses()) {
1005  bool recursive = isRecursiveClause(*clause);
1006  visitDepthFirst(*clause, [&](const AstAtom& atom) {
1007  if (atom.getName() == clause->getHead()->getName()) {
1008  return;
1009  }
1010 
1011  if (!isExistentialAtom(atom)) {
1012  if (recursive) {
1013  // Clause is recursive, so add an edge to the dependency graph
1014  relationGraph.insert(clause->getHead()->getName(), atom.getName());
1015  } else {
1016  // Non-existential apperance in a non-recursive clause, so
1017  // it's out of the picture
1018  minimalIrreducibleRelations.insert(atom.getName());
1019  }
1020  }
1021  });
1022  }
1023  }
1024 
1025  // TODO (see issue #564): Don't transform relations appearing in aggregators
1026  // due to aggregator issues with unnamed variables.
1027  visitDepthFirst(program, [&](const AstAggregator& aggr) {
1028  visitDepthFirst(
1029  aggr, [&](const AstAtom& atom) { minimalIrreducibleRelations.insert(atom.getName()); });
1030  });
1031 
1032  // Run a DFS from each 'bad' source
1033  // A node is reachable in a DFS from an irreducible node if and only if it is
1034  // also an irreducible node
1035  std::set<AstRelationIdentifier> irreducibleRelations;
1036  for (AstRelationIdentifier relationName : minimalIrreducibleRelations) {
1037  relationGraph.visitDepthFirst(relationName,
1038  [&](const AstRelationIdentifier& subRel) { irreducibleRelations.insert(subRel); });
1039  }
1040 
1041  // All other relations are necessarily existential
1042  std::set<AstRelationIdentifier> existentialRelations;
1043  for (AstRelation* relation : program.getRelations()) {
1044  if (!relation->getClauses().empty() &&
1045  irreducibleRelations.find(relation->getName()) == irreducibleRelations.end()) {
1046  existentialRelations.insert(relation->getName());
1047  }
1048  }
1049 
1050  // Reduce the existential relations
1051  for (AstRelationIdentifier relationName : existentialRelations) {
1052  AstRelation* originalRelation = program.getRelation(relationName);
1053 
1054  std::stringstream newRelationName;
1055  newRelationName << "+?exists_" << relationName;
1056 
1057  auto newRelation = std::make_unique<AstRelation>();
1058  newRelation->setName(newRelationName.str());
1059  newRelation->setSrcLoc(originalRelation->getSrcLoc());
1060 
1061  // EqRel relations require two arguments, so remove it from the qualifier
1062  newRelation->setQualifier(originalRelation->getQualifier() & ~(EQREL_RELATION));
1063 
1064  // Keep all non-recursive clauses
1065  for (AstClause* clause : originalRelation->getClauses()) {
1066  if (!isRecursiveClause(*clause)) {
1067  auto newClause = std::make_unique<AstClause>();
1068 
1069  newClause->setSrcLoc(clause->getSrcLoc());
1070  if (const AstExecutionPlan* plan = clause->getExecutionPlan()) {
1071  newClause->setExecutionPlan(std::unique_ptr<AstExecutionPlan>(plan->clone()));
1072  }
1073  newClause->setGenerated(clause->isGenerated());
1074  newClause->setFixedExecutionPlan(clause->hasFixedExecutionPlan());
1075  newClause->setHead(std::make_unique<AstAtom>(newRelationName.str()));
1076  for (AstLiteral* lit : clause->getBodyLiterals()) {
1077  newClause->addToBody(std::unique_ptr<AstLiteral>(lit->clone()));
1078  }
1079 
1080  newRelation->addClause(std::move(newClause));
1081  }
1082  }
1083 
1084  program.appendRelation(std::move(newRelation));
1085  }
1086 
1087  // Mapper that renames the occurrences of marked relations with their existential
1088  // counterparts
1089  struct M : public AstNodeMapper {
1090  const std::set<AstRelationIdentifier>& relations;
1091 
1092  M(std::set<AstRelationIdentifier>& relations) : relations(relations) {}
1093 
1094  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const override {
1095  if (AstClause* clause = dynamic_cast<AstClause*>(node.get())) {
1096  if (relations.find(clause->getHead()->getName()) != relations.end()) {
1097  // Clause is going to be removed, so don't rename it
1098  return node;
1099  }
1100  } else if (AstAtom* atom = dynamic_cast<AstAtom*>(node.get())) {
1101  if (relations.find(atom->getName()) != relations.end()) {
1102  // Relation is now existential, so rename it
1103  std::stringstream newName;
1104  newName << "+?exists_" << atom->getName();
1105  return std::make_unique<AstAtom>(newName.str());
1106  }
1107  }
1108  node->apply(*this);
1109  return node;
1110  }
1111  };
1112 
1113  M update(existentialRelations);
1114  program.apply(update);
1115 
1116  bool changed = !existentialRelations.empty();
1117  return changed;
1118 }

1120 bool NormaliseConstraintsTransformer::transform(AstTranslationUnit& translationUnit) {
1121  bool changed = false;
1122 
1123  // set a prefix for variables bound by magic-set for identification later
1124  // prepended by + to avoid conflict with user-defined variables
1125  static constexpr const char* boundPrefix = "+abdul";
1126 
1127  AstProgram& program = *translationUnit.getProgram();
1128 
1129  /* Create a node mapper that recursively replaces all constants and underscores
1130  * with named variables.
1131  *
1132  * The mapper keeps track of constraints that should be added to the original
1133  * clause it is being applied on in a given constraint set.
1134  */
1135  struct M : public AstNodeMapper {
1136  std::set<AstConstraint*>& constraints;
1137  mutable int changeCount;
1138 
1139  M(std::set<AstConstraint*>& constraints, int changeCount)
1140  : constraints(constraints), changeCount(changeCount) {}
1141 
1142  bool hasChanged() const {
1143  return changeCount > 0;
1144  }
1145 
1146  int getChangeCount() const {
1147  return changeCount;
1148  }
1149 
1150  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const override {
1151  if (AstStringConstant* stringConstant = dynamic_cast<AstStringConstant*>(node.get())) {
1152  // string constant found
1153  changeCount++;
1154 
1155  // create new variable name (with appropriate suffix)
1156  std::string constantValue = stringConstant->getConstant();
1157  std::stringstream newVariableName;
1158  newVariableName << boundPrefix << changeCount << "_" << constantValue << "_s";
1159 
1160  // create new constraint (+abdulX = constant)
1161  auto newVariable = std::make_unique<AstVariable>(newVariableName.str());
1162  constraints.insert(new AstConstraint(BinaryConstraintOp::EQ,
1163  std::unique_ptr<AstArgument>(newVariable->clone()),
1164  std::unique_ptr<AstArgument>(stringConstant->clone())));
1165 
1166  // update constant to be the variable created
1167  return std::move(newVariable);
1168  } else if (AstNumberConstant* numberConstant = dynamic_cast<AstNumberConstant*>(node.get())) {
1169  // number constant found
1170  changeCount++;
1171 
1172  // create new variable name (with appropriate suffix)
1173  AstDomain constantValue = numberConstant->getIndex();
1174  std::stringstream newVariableName;
1175  newVariableName << boundPrefix << changeCount << "_" << constantValue << "_n";
1176 
1177  // create new constraint (+abdulX = constant)
1178  auto newVariable = std::make_unique<AstVariable>(newVariableName.str());
1179  constraints.insert(new AstConstraint(BinaryConstraintOp::EQ,
1180  std::unique_ptr<AstArgument>(newVariable->clone()),
1181  std::unique_ptr<AstArgument>(numberConstant->clone())));
1182 
1183  // update constant to be the variable created
1184  return std::move(newVariable);
1185  } else if (dynamic_cast<AstUnnamedVariable*>(node.get())) {
1186  // underscore found
1187  changeCount++;
1188 
1189  // create new variable name
1190  std::stringstream newVariableName;
1191  newVariableName << "+underscore" << changeCount;
1192 
1193  return std::make_unique<AstVariable>(newVariableName.str());
1194  }
1195 
1196  node->apply(*this);
1197  return node;
1198  }
1199  };
1200 
1201  int changeCount = 0; // number of constants and underscores seen so far
1202 
1203  // apply the change to all clauses in the program
1204  for (AstRelation* rel : program.getRelations()) {
1205  for (AstClause* clause : rel->getClauses()) {
1206  if (clause->isFact()) {
1207  continue; // don't normalise facts
1208  }
1209 
1210  std::set<AstConstraint*> constraints;
1211  M update(constraints, changeCount);
1212  clause->apply(update);
1213 
1214  changeCount = update.getChangeCount();
1215  changed = changed || update.hasChanged();
1216 
1217  for (AstConstraint* constraint : constraints) {
1218  clause->addToBody(std::unique_ptr<AstConstraint>(constraint));
1219  }
1220  }
1221  }
1222 
1223  return changed;
1224 }
